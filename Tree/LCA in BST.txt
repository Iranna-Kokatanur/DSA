class BST
{   
    //Function to find the lowest common ancestor in a BST. 
	Node LCA(Node root, int n1, int n2)
	{
        if(root == null)  return null;
        int cur = root.data;
        
        if(cur > n1 && cur > n2) return LCA(root.left, n1, n2);
        if(cur < n1 && cur < n2) return LCA(root.right, n1, n2);
        
        return root;
    }
    
}
